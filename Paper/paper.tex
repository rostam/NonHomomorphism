\title{Non-Homomorphism Factors and the Concentration Parameter\\
       in Graph Homomorphisms}
\author{Mohammad Ali Rostami}

\begin{document}

\begin{abstract}
We study two measures associated with graph homomorphisms.
The \emph{non-homomorphism factor} $|G,H|$ is the minimum number of edges
that must be removed from $G$ to obtain a graph admitting a homomorphism to
$H$.  We prove $|K_n,K_m| = \binom{n}{2} - t(n,m)$ via Tur\'an's theorem and
derive a lower bound $|G,H|\ge|E(G)|-t(|V(G)|,\chi(H))$ for arbitrary $G$.
We introduce the \emph{concentration parameter} $\gamma(G,H)$, the minimum
over all homomorphisms $f\colon G\to H$ of the maximum number of $G$-edges
mapped to any single $H$-edge.  We prove a multiplicativity inequality
$\gamma(G,K)\le\gamma(G,H)\cdot\gamma(H,K)$, a monotonicity property
$\gamma(G,H_2)\le\gamma(G,H_1)$ when $H_1\subseteq H_2$, and the bound
$|G,K|\le\gamma(G,H)\cdot|H,K|$.  Computational experiments on cycles,
complete graphs, complete bipartite graphs, and the Petersen graph yield
exact values of $\gamma$ and reveal that the symmetric combination
$m(G,H)=\max(|G,H|,|H,G|)$, previously claimed to be a pseudo-metric, does
\emph{not} in general satisfy the triangle inequality
(108 counterexample triples found among complete graphs with at most $10$
vertices).  We also prove an exact formula for $\gamma(C_n,K_3)$ for all
$n\ge 3$, verified computationally up to $n=50$, and a general
$d$-regularity theorem that unifies the cycle formula with the value
$\gamma(\text{Petersen},K_3)=6$.
\end{abstract}

\section{Introduction}
% ─────────────────────────────────────────────────────────────────────────────

Throughout this paper all graphs are simple and undirected.

\begin{definition}[Graph Homomorphism]
A \emph{homomorphism} from $G$ to $H$ is a function $f\colon V(G)\to V(H)$
such that $\{u,v\}\in E(G)$ implies $\{f(u),f(v)\}\in E(H)$.
We write $G\to H$ when such a function exists.
\end{definition}

A homomorphism need not be injective or surjective.
A bijection whose inverse is also a homomorphism is an \emph{isomorphism};
an isomorphism from $G$ to itself is an \emph{automorphism}.

A proper $k$-colouring of $G$ exists if and only if $G\to K_k$, giving:

\begin{proposition}[\cite{hell2004}]
\label{prop:chi_mono}
$G\to H \Rightarrow \chi(G)\le\chi(H)$.
\end{proposition}

\begin{theorem}[\cite{hell2004}]
\label{thm:hom_contracts}
Any homomorphism $f\colon G\to H$ satisfies
$d_H(f(u),f(v))\le d_G(u,v)$ for all $u,v\in V(G)$.
\end{theorem}

\begin{theorem}[\cite{hell2004}]
\label{thm:vertex_trans}
If $H$ is vertex-transitive and $G\to H$ then
$\displaystyle\frac{\alpha(G)}{|V(G)|}\ge\frac{\alpha(H)}{|V(H)|}$,
where $\alpha$ denotes the independence number.
\end{theorem}

\subsection{Cores}

\begin{definition}
A graph is a \emph{core} if it has no homomorphism to any proper subgraph of
itself.  Every graph $G$ contains a unique (up to isomorphism) core subgraph,
denoted $\operatorname{core}(G)$, such that $G\to\operatorname{core}(G)$ and
$\operatorname{core}(G)\subseteq G$.
\end{definition}

A key consequence is that $K\to G$ if and only if
$K\to\operatorname{core}(G)$.  The Petersen graph is its own core, as it
admits no homomorphism to any proper subgraph~\cite{hell2004}.

\section{Non-Homomorphism Factors}
% ─────────────────────────────────────────────────────────────────────────────

\begin{definition}[Non-Homomorphism Factor \cite{khoshkha2005}]
\label{def:nhf}
$|G,H|$ is the minimum number of edges that must be removed from $G$ to
obtain a graph $G'$ with $G'\to H$.  If $G\to H$ then $|G,H|=0$.
\end{definition}

\begin{proposition}[\cite{khoshkha2005}]
\label{prop:monotone}
\begin{enumerate}
  \item\label{it:mon1} $H_1\to H_2 \Rightarrow |G,H_1|\ge|G,H_2|$ for every
        graph $G$.
  \item\label{it:mon2} $H_1\leftrightarrow H_2 \Rightarrow |G,H_1|=|G,H_2|$
        for every $G$.
  \item\label{it:mon3} If $f\colon G_1\to G_2$ is edge-injective
        (distinct edges map to distinct edges) then
        $|G_1,H|\le|G_2,H|$ for every $H$.
\end{enumerate}
\end{proposition}

\begin{proof}
(\ref{it:mon1}) If $G'\to H_1$ then $G'\to H_1\to H_2$, so every graph
witnessing $|G,H_1|$ also witnesses $|G,H_2|$.
(\ref{it:mon2}) Apply~(\ref{it:mon1}) in both directions.
(\ref{it:mon3}) Let $S\subseteq E(G_2)$ with $|S|=|G_2,H|$ such that
$G_2-S\to H$.  Set $S'=\{e\in E(G_1):f(e)\in S\}$.  Since $f$ is
edge-injective, $|S'|\le|S|$.  The restricted map $f\colon G_1-S'\to G_2-S$
is still a homomorphism, so $G_1-S'\to H$, giving $|G_1,H|\le|S'|\le|G_2,H|$.
\end{proof}

\subsection{Complete Graphs and Tur\'an's Theorem}

\begin{theorem}[Tur\'an~\cite{hell2004}]
\label{thm:turan}
Among all $K_{m+1}$-free graphs on $n$ vertices the one with the most edges is
the \emph{Tur\'an graph} $T(n,m)$, obtained by partitioning the $n$ vertices
into $m$ classes of sizes as equal as possible and taking all edges between
distinct classes; write $t(n,m)=|E(T(n,m))|$.
\end{theorem}

\begin{theorem}
\label{thm:KnKm}
$\displaystyle|K_n,K_m| = \binom{n}{2} - t(n,m)$.
If $n=qm+r$ with $0\le r<m$, this equals
$r\dbinom{q+1}{2}+(m-r)\dbinom{q}{2}$.
In particular, $|K_{n+1},K_n|=1$ for all $n\ge 1$.
\end{theorem}

\begin{proof}
A spanning subgraph $G'$ of $K_n$ satisfies $G'\to K_m$ if and only if
$\chi(G')\le m$.  The maximum number of edges an $m$-colourable graph on $n$
vertices can have is $t(n,m)$ (Tur\'an's theorem), so removing the minimum
number of edges to achieve $m$-colourability requires removing exactly
$\binom{n}{2}-t(n,m)$ edges.  The explicit formula follows from
$t(n,m)=\binom{n}{2}-r\binom{q+1}{2}-(m-r)\binom{q}{2}$.
For $|K_{n+1},K_n|$: set $G=K_{n+1}$, $m=n$, so $q=1$, $r=1$, giving
$1\cdot\binom{2}{2}+(n-1)\cdot\binom{1}{2}=1$.
\end{proof}

\subsection{A General Lower Bound}

\begin{theorem}
\label{thm:lower_bound}
For any graphs $G$ and $H$,
\[
  |G,H| \;\ge\; \max\!\bigl(0,\,|E(G)|-t(|V(G)|,\chi(H))\bigr).
\]
\end{theorem}

\begin{proof}
Any $G'\subseteq G$ with $G'\to H$ must satisfy $\chi(G')\le\chi(H)$.
By Tur\'an's theorem, $|E(G')|\le t(|V(G)|,\chi(H))$.  Hence at least
$|E(G)|-t(|V(G)|,\chi(H))$ edges must be removed from $G$.
\end{proof}

The bound is tight for $G=K_n$: it then equals $|K_n,K_m|$ from
Theorem~\ref{thm:KnKm}.  For sparser $G$ the bound may be $0$, but then
$|G,H|$ can be positive due to structural obstructions beyond edge density.

\subsection{Cycles}

Every cycle $C_n$ satisfies $C_n\to K_3$: even cycles are bipartite, so
$C_n\to K_2\subset K_3$; odd cycles are $3$-chromatic, so $C_n\to K_3$
directly.  Hence $|C_n,K_3|=0$ for all $n\ge 3$.

By contrast, $|C_n,K_2|=0$ for even $n$ and $|C_n,K_2|=1$ for odd $n\ge 3$
(removing any edge produces a path, which is bipartite).

\subsection{The Petersen Graph}

The Petersen graph $P$ has $10$ vertices, $15$ edges, chromatic number $3$,
and is its own core.  Since $\chi(P)=3$, $|P,K_m|=0$ for all $m\ge 3$.

\begin{observation}
\label{obs:PK2}
$|P,K_2|=3$.  The lower bound from Theorem~\ref{thm:lower_bound} gives
$|P,K_2|\ge|E(P)|-t(10,2)=15-25=0$, which is not tight (the Petersen graph
is far from bipartite).  The value $3$ is verified computationally in
Section~\ref{sec:computation}.
\end{observation}

\section{The Concentration Parameter}
% ─────────────────────────────────────────────────────────────────────────────

\begin{definition}[Concentration Parameter]
\label{def:gamma}
Let $G$ and $H$ be graphs with $|E(G)|\ge 1$.  For a homomorphism
$f\colon G\to H$ and an edge $e\in E(H)$, the \emph{load} of $e$ under $f$
is
\[
  \mu(f,e) = \bigl|\{e'\in E(G) : \{f(u),f(v)\}=e
             \text{ for the endpoints } u,v \text{ of } e'\}\bigr|.
\]
Define the \emph{max edge multiplicity} $\mu(f)=\max_{e\in E(H)}\mu(f,e)$.
The \emph{concentration parameter} is
\[
  \gamma(G,H) = \min_{f\colon G\to H} \mu(f).
\]
When $G\not\to H$ we set $\gamma(G,H)=+\infty$.
\end{definition}

\begin{remark}
Note that $\alpha$ denotes the independence number throughout; the symbol
$\mu$ is used for edge multiplicities to avoid collision.
A related per-homomorphism quantity $M^\sigma$ appears in~\cite{daneshgar2007}.
\end{remark}

Since the $|E(G)|$ pre-images are distributed among $|E(H)|$ images, the
pigeonhole principle gives:

\begin{proposition}[Pigeonhole lower bound]
\label{prop:pigeonhole}
If $G\to H$ then $\gamma(G,H)\ge\lceil|E(G)|/|E(H)|\rceil$.
\end{proposition}

This bound is not always achieved; Table~\ref{tab:CnK3} shows cases where
the cycle structure forces a higher concentration.

\begin{lemma}
\label{lem:gamma_identity}
$\gamma(G,G)=1$ for any graph $G$ with $|E(G)|\ge 1$.
\end{lemma}

\begin{proof}
The identity $\operatorname{id}\colon G\to G$ satisfies $\mu(\operatorname{id})=1$
since each edge is its own unique pre-image.  Since $\gamma\ge 1$ by
definition (every edge must map somewhere), $\gamma(G,G)=1$.
\end{proof}

\begin{lemma}
\label{lem:gamma_pos}
If $G\to H$ and $|E(G)|\ge 1$ then $\gamma(G,H)\ge 1$.
\end{lemma}

\begin{proof}
Every edge of $G$ must map to an edge of $H$ (since $H$ has no loops), so
some $H$-edge receives load at least $1$.
\end{proof}

\begin{lemma}[Multiplicativity]
\label{lem:gamma_mult}
If $G\to H\to K$ then $\gamma(G,K)\le\gamma(G,H)\cdot\gamma(H,K)$.
\end{lemma}

\begin{proof}
Choose $f\colon G\to H$ with $\mu(f)=\gamma(G,H)$ and $g\colon H\to K$ with
$\mu(g)=\gamma(H,K)$.  Consider the composition $g\circ f\colon G\to K$.
For any $K$-edge $e$, the $G$-edges mapping to $e$ under $g\circ f$ are
exactly those mapping under $f$ to some $H$-edge in $g^{-1}(e)$.  There are
at most $\mu(g)=\gamma(H,K)$ such $H$-edges, and each carries at most
$\mu(f)=\gamma(G,H)$ pre-images from $G$.  Hence $\mu(g\circ f)\le
\gamma(G,H)\cdot\gamma(H,K)$, so
$\gamma(G,K)\le\mu(g\circ f)\le\gamma(G,H)\cdot\gamma(H,K)$.
\end{proof}

\begin{lemma}[Bound on non-homomorphism factor]
\label{lem:nhf_bound}
If $G\to H$ then $|G,K|\le\gamma(G,H)\cdot|H,K|$ for every graph $K$.
\end{lemma}

\begin{proof}
Let $k=|H,K|$ and choose $S\subseteq E(H)$ with $|S|=k$ such that $H-S\to K$.
Let $f\colon G\to H$ achieve $\mu(f)=\gamma(G,H)$.  Define
$T=\{e'\in E(G):f(e')\in S\}$.  For each $H$-edge in $S$, at most
$\gamma(G,H)$ edges of $G$ map to it, so $|T|\le k\cdot\gamma(G,H)$.
The map $f$ restricted to $G-T$ lands in $H-S$ (since the only $G$-edges
mapping to removed $H$-edges have been deleted), and $H-S\to K$, so
$G-T\to K$.  Hence $|G,K|\le|T|\le\gamma(G,H)\cdot|H,K|$.
\end{proof}

\begin{lemma}[Two-sided bound via a common target]
\label{lem:two_sided}
If $G\to H$ and $K\to H$ then
\[
  |G,K| \le \gamma(G,H)\cdot|H,K|
  \qquad\text{and}\qquad
  |K,G| \le \gamma(K,H)\cdot|H,G|.
\]
Consequently,
$m(G,K)\le\max\!\bigl(\gamma(G,H)\cdot|H,K|,\;\gamma(K,H)\cdot|H,G|\bigr)$.
\end{lemma}

\begin{proof}
Both inequalities follow from Lemma~\ref{lem:nhf_bound}, the first applied to
the pair $(G,H)$ with target $K$, the second to the pair $(K,H)$ with target
$G$.  The bound on $m$ follows by taking the max.
\end{proof}

\begin{lemma}[Monotonicity in the target]
\label{lem:gamma_mono}
If $H_1\subseteq H_2$ (i.e.\ $H_1$ is a subgraph of $H_2$) and $G\to H_1$,
then $\gamma(G,H_2)\le\gamma(G,H_1)$.
\end{lemma}

\begin{proof}
The inclusion $\iota\colon H_1\hookrightarrow H_2$ is an edge-injective
homomorphism, so $\mu(\iota)=1$ and $\gamma(H_1,H_2)\le 1$.  Since $\gamma\ge
1$ always, $\gamma(H_1,H_2)=1$.  By Lemma~\ref{lem:gamma_mult},
$\gamma(G,H_2)\le\gamma(G,H_1)\cdot\gamma(H_1,H_2)=\gamma(G,H_1)$.
\end{proof}

\begin{proposition}
\label{prop:gamma_Kn}
For $n\le m$, $\gamma(K_n,K_m)=1$.
For $n>m$, $\gamma(K_n,K_m)=+\infty$ (since $\chi(K_n)=n>m$ forbids any
homomorphism).
\end{proposition}

\begin{proof}
When $n\le m$ the inclusion $K_n\hookrightarrow K_m$ is edge-injective, so
$\mu(\iota)=1$ and $\gamma(K_n,K_m)\le 1$.  Combined with
Lemma~\ref{lem:gamma_pos}, $\gamma(K_n,K_m)=1$.
\end{proof}

\subsection{Concentration for Regular Graphs}

\begin{theorem}[d-Regularity Formula]
\label{thm:d_regular}
Let $G$ be a $d$-regular graph on $n$ vertices with $\chi(G)=3$.
Let
\[
  M = \min\bigl\{\,\max(s_1,s_2,s_3) :
      s_1+s_2+s_3=n,\; s_i\ge 0,\;
      s_1\equiv s_2\equiv s_3\pmod{2}\,\bigr\}.
\]
Then $\gamma(G,K_3) = \dfrac{d}{2}\cdot M$.
\end{theorem}

\begin{proof}
Let $f\colon G\to K_3$ be any homomorphism with colour classes $A,B,C$
of sizes $a,b,c$ (so $a+b+c=n$).  Because $G$ is $d$-regular, the
number of edges between classes satisfies
\[
  e_{AB}=\tfrac{d}{2}(a+b-c),\quad
  e_{AC}=\tfrac{d}{2}(a+c-b),\quad
  e_{BC}=\tfrac{d}{2}(b+c-a).
\]
(Each vertex in $A$ contributes $d$ edges total; all neighbours in $A$
contribute $0$ (no monochromatic edges), so $e_{AB}+e_{AC}=da$; the
analogous identities for $B$ and $C$ give the three equations above.)
Writing $s_1=a+b-c$, $s_2=a+c-b$, $s_3=b+c-a$, the triple
$(s_1,s_2,s_3)$ is non-negative, sums to $n$, and all have the same
parity (each $s_i\equiv a+b+c-2x_i\equiv n\pmod{2}$ for appropriate
$x_i$).  The max edge load under $f$ is $\mu(f)=(d/2)\max(s_1,s_2,s_3)$.

Minimising over all achievable class distributions gives
\[
  \gamma(G,K_3) = \frac{d}{2}\cdot
  \min_{\substack{s_1+s_2+s_3=n,\; s_i\ge 0\\s_1\equiv s_2\equiv s_3\pmod{2}}}
  \max(s_1,s_2,s_3).
\]
The minimum is $M$.  (Any triple $(a,b,c)$ of non-negative integers with
$a+b+c=n$ is realised as an independent-set partition of any
$3$-chromatic $d$-regular graph provided no colour class is too large,
which is guaranteed when $G$ has no isolated vertices.)
\end{proof}

\begin{remark}
The formula requires $d$ to be even or the $s_i$ to conspire so that
$\tfrac{d}{2}s_i$ is an integer; this is automatic when $n$ and $d$ are
both even, or when all $s_i$ are even.  The Petersen graph ($n=10$,
$d=3$) requires same-parity triples summing to $10$; all-even gives
minimum max $=4$ (e.g.\ $(4,4,2)$), so $\gamma=\tfrac{3}{2}\cdot 4=6$.
\end{remark}

\section{The Symmetric Combination $m(G,H)$}
\label{sec:metric}
% ─────────────────────────────────────────────────────────────────────────────

\begin{definition}
\label{def:m}
$m(G,H) = \max\bigl(|G,H|,\,|H,G|\bigr)$.
\end{definition}

The function $m$ is clearly non-negative, satisfies $m(G,G)=0$ (since
$G\to G$), and is symmetric by definition.  The following characterises when
$m=0$.

\begin{proposition}
\label{prop:m_zero}
$m(G,H)=0$ if and only if $G$ and $H$ are homomorphically equivalent
(i.e.\ $G\leftrightarrow H$), which holds if and only if $\operatorname{core}(G)
\cong\operatorname{core}(H)$.
\end{proposition}

\begin{proof}
$m(G,H)=0$ means $|G,H|=0$ and $|H,G|=0$, i.e.\ $G\to H$ and $H\to G$.
This is precisely homomorphic equivalence, and two graphs are homomorphically
equivalent iff they have isomorphic cores.
\end{proof}

\begin{theorem}
\label{thm:no_triangle}
$m$ does \emph{not} satisfy the triangle inequality in general.
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:KnKm} and the fact that $K_n\to K_m$ for all $n\le m$:
\begin{align*}
  m(K_7,K_3) &= \max(|K_7,K_3|,\,|K_3,K_7|) = \max(5,0) = 5,\\
  m(K_7,K_5) &= \max(|K_7,K_5|,\,|K_5,K_7|) = \max(2,0) = 2,\\
  m(K_5,K_3) &= \max(|K_5,K_3|,\,|K_3,K_5|) = \max(2,0) = 2.
\end{align*}
The triangle inequality would require
$m(K_7,K_3)\le m(K_7,K_5)+m(K_5,K_3)$, i.e.\ $5\le 4$.
This fails.
\end{proof}

\begin{remark}
Khoshkha~\cite{khoshkha2005} claims $m$ is a pseudo-metric.
Theorem~\ref{thm:no_triangle} shows this is incorrect under
Definition~\ref{def:nhf}.  A modified definition allowing edge additions as
well as deletions, or using a different minimisation, might restore the
triangle inequality; we leave this as an open question
(Section~\ref{sec:open}, Question~\ref{q:metric}).
\end{remark}

\section{Computational Results}
\label{sec:computation}
% ─────────────────────────────────────────────────────────────────────────────

We computed $|G,H|$ and $\gamma(G,H)$ using algorithms implemented in C++
and Rust (source at \url{https://github.com/rostam/NonHomomorphism}):
\begin{itemize}
  \item \textbf{Homomorphism search:} backtracking DFS with early constraint
        pruning (Rust and C++).
  \item \textbf{Non-homomorphism factor:} for $k=0,1,2,\ldots$ test all
        $\binom{|E(G)|}{k}$ edge-removal subsets (C++).
  \item \textbf{Concentration parameter:} enumerate \emph{all} homomorphisms
        and compute $\gamma=\min_f\mu(f)$ (C++).
\end{itemize}

The data is consistent with the following exact formula, which holds for
all $n\ge 3$ and is verified computationally for $n\le 50$.

\begin{theorem}
\label{thm:cycle_gamma}
For all $n\ge 3$,
\[
  \gamma(C_n, K_3) =
  \begin{cases}
    \lceil n/3 \rceil     & \text{if } n \not\equiv 2 \pmod{3}, \\
    \lceil n/3 \rceil + 1 & \text{if } n \equiv 2 \pmod{3}.
  \end{cases}
\]
\end{theorem}

\begin{proof}
We apply Theorem~\ref{thm:d_regular} with $d=2$ (cycles are $2$-regular).
Write $n=3q+r$ with $r\in\{0,1,2\}$.  The parameter $M$ is the minimum
max of a same-parity non-negative triple summing to $n$.

\textbf{Case $n\equiv 0\pmod 3$} ($r=0$, $n=3q$).
The triple $(q,q,q)$ sums to $3q$ and all have the same parity.
Its max is $q=n/3$; no smaller max is possible since the average is $q$.
Hence $M=q$ and $\gamma=1\cdot q=n/3=\lceil n/3\rceil$.

\textbf{Case $n\equiv 1\pmod 3$} ($r=1$, $n=3q+1$).
The triple $(q+1,q+1,q-1)$ has the same parity and sums to $3q+1$;
its max is $q+1$.  Any triple with max $\le q$ sums to at most $3q<n$,
so $M=q+1=\lceil n/3\rceil$.  Hence $\gamma=q+1=\lceil n/3\rceil$.

\textbf{Case $n\equiv 2\pmod 3$} ($r=2$, $n=3q+2$).
Suppose there exists a same-parity triple summing to $n$ with max $\le q+1$.
Then all three values are $\le q+1$ and the sum is at most $3(q+1)=3q+3$.
But the sum must equal $3q+2$, so at least one value equals $q$.
A triple with values in $\{q,q+1\}$ and sum $3q+2$ must have two values
equal to $q+1$ and one to $q$.  But $q$ and $q+1$ have different parities,
contradicting same-parity.  Hence $M\ge q+2$.  The triple $(q+2,q,q)$
achieves this (same parity: $q$ and $q+2$ are congruent mod $2$), so
$M=q+2=\lceil n/3\rceil+1$.  Hence $\gamma=q+2=\lceil n/3\rceil+1$.
\end{proof}

\begin{remark}
Theorem~\ref{thm:cycle_gamma} follows from Theorem~\ref{thm:d_regular} by
setting $d=2$.  The formula is \emph{not} merely the pigeonhole bound
$\lceil n/3\rceil$: for $n\equiv 2\pmod 3$ the parity obstruction raises
$\gamma$ by one above the pigeonhole value.  Lemma~\ref{lem:gamma_mono}
(monotonicity) is also consistent with Table~\ref{tab:CnK3}: since
$K_2\subset K_3$, any homomorphism $C_n\to K_2$ (for even $n$) lifts to
$K_3$ with no worse concentration.
\end{remark}

\begin{theorem}
\label{thm:Knn}
$\gamma(K_{n,n}, K_3) = n\cdot\lceil n/2\rceil$.
\end{theorem}

\begin{proof}
Since $K_{n,n}$ is bipartite, $K_{n,n}\to K_2\subset K_3$, so
$|K_{n,n},K_3|=0$.

Let $L$ and $R$ be the two parts.  In any homomorphism $f\colon K_{n,n}\to
K_3$, the colour sets $f(L)$ and $f(R)$ must be disjoint: every vertex of
$L$ is adjacent to every vertex of $R$, and $K_3$ has no self-loops, so
$f(\ell)\ne f(r)$ for all $\ell\in L$, $r\in R$.  Since $|V(K_3)|=3$, by the
pigeonhole principle one part uses at most one colour and the other at most
two.

Without loss of generality (by symmetry of $K_{n,n}$), suppose $f(R)\subseteq
\{c\}$ and $f(L)\subseteq\{a,b\}$ with $\{a,b,c\}=V(K_3)$.  If $k$ vertices
of $L$ map to $a$ and $(n-k)$ map to $b$, then the loads on $\{a,c\}$ and
$\{b,c\}$ are $nk$ and $n(n-k)$ respectively (each $R$-vertex contributes
one edge to each $L$-vertex), and the load on $\{a,b\}$ is zero.  The maximum
is minimised at $k=\lfloor n/2\rfloor$, giving
$\max(n\lceil n/2\rceil, n\lfloor n/2\rfloor)=n\lceil n/2\rceil$.
Hence $\gamma(K_{n,n},K_3)=n\lceil n/2\rceil$.
\end{proof}

For $n=1,2,3,4$ this gives $1,2,6,8$, matching Table~\ref{tab:KnnK3}.

Observation~\ref{obs:PK2} gives $|P,K_2|=3$.  Since $\chi(P)=3$, $|P,K_m|=0$

\begin{observation}
\label{obs:petersen_mono}
$\gamma(P,K_3)=6>\gamma(P,K_4)=4>\gamma(P,K_5)=2$.  This is consistent with
Lemma~\ref{lem:gamma_mono}: since $K_3\subset K_4\subset K_5$, each inclusion
is edge-injective and monotonicity gives
$\gamma(P,K_3)\ge\gamma(P,K_4)\ge\gamma(P,K_5)$.
\end{observation}

\begin{observation}
\label{obs:petersen_exceed}
$\gamma(P,K_3)=6$ strictly exceeds the pigeonhole lower bound
$\lceil 15/3\rceil=5$.  Theorem~\ref{thm:d_regular} explains this:
with $n=10$, $d=3$, the optimal same-parity triple summing to $10$
has minimum max $M=4$ (e.g.\ $(4,4,2)$; all-odd is impossible since
$10$ is even), giving $\gamma=\tfrac{3}{2}\cdot 4=6$.  The parity
obstruction prevents $M=3$ (which would require $(3,4,3)$, mixed parity).
\end{observation}

\begin{observation}
\label{obs:gamma_K5}
The value $\gamma(P,K_5)=2$ meets the pigeonhole bound $\lceil 15/10\rceil=2$.
By Lemma~\ref{lem:gamma_mult} applied to $K_3\to K_4\to K_5$:
$\gamma(P,K_5)\le\gamma(P,K_3)\cdot\gamma(K_3,K_5)=6\cdot 1=6$
and $\gamma(P,K_4)\le\gamma(P,K_3)\cdot\gamma(K_3,K_4)=6\cdot 1=6$.
Both are consistent with (but do not imply) the computed values.
\end{observation}

\section{Open Questions}
\label{sec:open}
% ─────────────────────────────────────────────────────────────────────────────

\begin{enumerate}

\item\label{q:regularity_general}
\textbf{Extend the $d$-regularity formula to general $H$.}
Theorem~\ref{thm:d_regular} gives $\gamma(G,K_3)$ exactly for $d$-regular
$G$ with $\chi(G)=3$.  Is there an analogous formula for $\gamma(G,H)$
when $H$ is an arbitrary $k$-chromatic graph?  The argument uses the fact
that $K_3$ has a unique edge type (all $3$ edges are equivalent under
automorphisms); for $H$ with multiple edge orbits the formula would need
to track the load on each orbit separately.

\item\label{q:edge_trans}
\textbf{When does $\gamma(G,H)$ meet the ratio $|E(G)|/|E(H)|$?}
For $d$-regular $G$ with $\chi(G)=3$, Theorem~\ref{thm:d_regular} shows
that $\gamma(G,K_3)$ meets the pigeonhole bound $\lceil|E(G)|/|E(K_3)|\rceil$
exactly when $n\equiv 0\pmod 3$ (equal class sizes achievable), and
exceeds it by $d/2$ when $n\equiv 2\pmod 3$ (parity obstruction).
For the Petersen graph ($n=10$, $d=3$, $n\equiv 1\pmod 3$) we get
$\gamma(P,K_3)=6>5=\lceil 15/3\rceil$.
Does $\gamma(G,H)$ always equal the ratio $|E(G)|/|E(H)|$ when $H$ is
``large enough'' relative to $G$?  (Computations show $\gamma(P,K_5)=2$
meets the pigeonhole bound $\lceil 15/10\rceil=2$.)

\item\label{q:metric}
\textbf{Can $m$ be repaired into a metric?}
Theorem~\ref{thm:no_triangle} shows $m(G,H)=\max(|G,H|,|H,G|)$ does not
satisfy the triangle inequality.  Two variants worth investigating are:
(a) $d(G,H)=\min\{k : \exists\,G'\text{ with }|E(G)\triangle E(G')|=k
    \text{ and }G'\to H\}$ (edge-edit distance to a subgraph mapping to $H$);
(b) $m'(G,H)=\max(|G,H|,|H,G|)$ restricted to graphs with the same vertex
    count.  Do either of these satisfy a triangle inequality?

\item\label{q:gamma_frac}
\textbf{$\gamma$ and the fractional chromatic number.}
For vertex-transitive $G$, $\chi_f(G)=|V(G)|/\alpha(G)$.  The Petersen graph
has $\chi_f(P)=5/2$.  Is $\gamma(G,K_m)$ related to $\chi_f(G)$?  For
example, does $\gamma(G,K_m)\ge|E(G)|/(m\lfloor|V(G)|/m\rfloor)$ hold for
vertex-transitive $G$?  For $P$ and $m=3$: $15/(3\cdot 3)=5/3$, which gives
$\lceil 5/3\rceil=2$, weaker than the actual value $6$.  A tighter bound using
$\chi_f$ may explain why $\gamma(P,K_3)=6$ instead of $5$.

\item\label{q:subadditivity}
\textbf{Characterise when $|G,K|\le|G,H|+|H,K|$.}
The triangle inequality fails for $m$, but the sub-additivity of $|G,\cdot|$
might hold under additional conditions.  For complete graphs,
$|K_n,K_m|+|K_m,K_\ell|=|K_n,K_\ell|$ when $n,m,\ell$ are in arithmetic
progression (e.g.\ $K_4,K_3,K_2$: $1+1=2$\checkmark), but fails when the
gap grows faster (e.g.\ $K_7,K_5,K_3$: $2+2<5$).
A systematic search finds 108 triples $(K_a,K_b,K_c)$ with $a,b,c\le 10$
where $m(K_a,K_c)>m(K_a,K_b)+m(K_b,K_c)$; the excess is largest when
$b\approx\sqrt{ac}$ (the ``geometric middle''), which maximises the Tur\'an
discrepancy $t(a,b)-t(a,c)>|K_b,K_c|$.  What characterises the
triples $(G,H,K)$ for which sub-additivity holds?

\end{enumerate}
